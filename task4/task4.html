panel.hist <- function(x, ...)    #i funkcija reikalinga grafik lentelei ibrti (histogramos pateikimui)                        
{
  usr <- par("usr"); on.exit(par(usr))
  par(usr = c(usr[1:2], 0, 1.5) )
  h <- hist(x, plot = FALSE)
  breaks <- h$breaks; nB <- length(breaks)
  y <- h$counts; y <- y/max(y)
  rect(breaks[-nB], 0, breaks[-1], y, col = "cyan", ...)
}

panel.cor <- function(x, y, digits = 2, prefix = "", cex.cor,...)    #i funkcija reikalinga grafik lentelei
    #ibrti (koreliacijos koeficiento radimui)
{
  usr <- par("usr"); on.exit(par(usr))
  par(usr = c(0, 1, 0, 1))
  r <- abs(cor(x, y))
  txt <- format(c(r, 0.123456789), digits = digits)[1]
  txt <- paste0(prefix, txt)
  if(missing(cex.cor)) cex.cor <- 0.8/strwidth(txt)
  text(0.5, 0.5, txt, cex = 3)
}

Toliau panaikiname iskirtis i duomen (funkcijos idja gauta i http://stackoverflow.com/questions/4787332/how-to-remove-outliers-from-a-dataset)

remove_outliers <- function(x) {              
  qnt <- quantile(x, probs=c(.25, .75), na.rm = T)       #na.rm logical; if true, any NA and NaN's are removed
   #from x before the quantiles are computed.  Randami 25% ir 75% kvantiliai
  H <- 1.5 * IQR(x, na.rm = T)      #randamas atstumas, kur reikia pridti prie 75% kvantilio ir atimti i 
   #25% kvantilio, kad rasti rib, nuo kurio galime teigti, kad duomenys u tos ribos bus iskirtys. 
  y <- x
  y[x < (qnt[1] - H)] <- NA                  # duomenys, perengianios nustatyt rib ir yra pakaiiami  NA
  y[x > (qnt[2] + H)] <- NA
  return(y)
}
