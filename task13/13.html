Funkcija skirta transformuoti duomenis ir ibrti transformuotus stacionarius ir pradinius duomenis

(Funkcij grafikas sukriau pats, dvi eiluts automatinio diferencijavimo buvo rastos fpp knygoje)

grafikas <-function(data, sez){
  par(mfrow=c(1,3))
  ns=0                                                #kad nebt klaidinama kiek kart differencijuota
  lambda = BoxCox.lambda(data)                        #randamas lambda  
  trdiff = BoxCox(data,lambda = lambda)               #transformuojami duomenys
  
  if(sez==T){                                         #naikinamas sezonikumo taka, jei yra sezonikumas
    ns <- nsdiffs(trdiff)
    if(ns>0)
      trdiff <- diff(trdiff,lag=frequency(trdiff), differences=ns)
  }
  
  nd <- ndiffs(trdiff)                                #diferencijuojama, jei trendas yra
  if(nd > 0)
    trdiff <- diff(trdiff,differences=nd)
  
  plot(data, main="pradinis")                            #pradini duomen grafikas
  plot(trdiff, ylab="pokytis", main= "stacionarus")     
  acf(trdiff, main = "nera nei sezoniskmo nei trendo")   #stacionari duomen grafikas
  p=kpss.test(trdiff)$p.value
  
    return(kable(data.frame(trendo_diferencijavimas = nd , sezono_diferencijavimas = ns, i_viso_diferencijuota = nd + ns,Lambda = lambda, p_reiksme=p)))            #atspausdinama statistin lentel
}
